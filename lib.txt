local library = {}
local UserInputService = game:GetService("UserInputService")

function library.new(title)
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "iqsmLibraryGui"
    screenGui.ResetOnSpawn = false
    screenGui.Parent = game.CoreGui

    local mainFrame = Instance.new("Frame")
    mainFrame.Size = UDim2.new(0, 480, 0, 320)
    mainFrame.Position = UDim2.new(0.5, -240, 0.5, -160)
    mainFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 45)
    mainFrame.Parent = screenGui

    local titleLabel = Instance.new("TextLabel")
    titleLabel.Text = title
    titleLabel.Size = UDim2.new(1, 0, 0, 50)
    titleLabel.BackgroundTransparency = 1
    titleLabel.Font = Enum.Font.GothamBold
    titleLabel.TextColor3 = Color3.fromRGB(200, 200, 255)
    titleLabel.TextSize = 26
    titleLabel.Parent = mainFrame

    local tabsFrame = Instance.new("Frame")
    tabsFrame.Size = UDim2.new(0, 120, 1, -50)
    tabsFrame.Position = UDim2.new(0, 0, 0, 50)
    tabsFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 35)
    tabsFrame.Parent = mainFrame

    local buttonsFrame = Instance.new("Frame")
    buttonsFrame.Size = UDim2.new(1, -120, 1, -50)
    buttonsFrame.Position = UDim2.new(0, 120, 0, 50)
    buttonsFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 30)
    buttonsFrame.Parent = mainFrame

    local tabs = {}
    local buttons = {}
    local currentTab = nil

    local function showTab(tabName)
        if currentTab == tabName then return end
        currentTab = tabName
        for bName, bList in pairs(buttons) do
            for _, btn in ipairs(bList) do
                btn.Visible = (bName == tabName)
            end
        end
    end

    function library:tab(name)
        if name == "iqsm." then return end
        if tabs[name] then return end

        local tabButton = Instance.new("TextButton")
        tabButton.Size = UDim2.new(1, 0, 0, 40)
        tabButton.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
        tabButton.TextColor3 = Color3.fromRGB(180, 180, 220)
        tabButton.Font = Enum.Font.Gotham
        tabButton.TextSize = 20
        tabButton.Text = name
        tabButton.Parent = tabsFrame

        table.insert(tabs, tabButton)
        buttons[name] = {}

        tabButton.MouseButton1Click:Connect(function()
            showTab(name)
        end)

        if not currentTab then
            showTab(name)
        end

        return name
    end

    function library:button(text, callback, tabName)
        tabName = tabName or currentTab
        if tabName == "iqsm." then return end
        if not buttons[tabName] then return end

        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(1, -20, 0, 35)
        btn.Position = UDim2.new(0, 10, 0, (#buttons[tabName] * 40) + 5)
        btn.BackgroundColor3 = Color3.fromRGB(60, 60, 90)
        btn.TextColor3 = Color3.fromRGB(220, 220, 255)
        btn.Font = Enum.Font.GothamSemibold
        btn.TextSize = 18
        btn.Text = text
        btn.Parent = buttonsFrame

        btn.MouseButton1Click:Connect(callback)

        table.insert(buttons[tabName], btn)
        return btn
    end

    function library:unload()
        local confirm = Instance.new("TextButton")
        confirm.Size = UDim2.new(0, 250, 0, 120)
        confirm.Position = UDim2.new(0.5, -125, 0.5, -60)
        confirm.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        confirm.TextColor3 = Color3.new(1, 1, 1)
        confirm.Font = Enum.Font.GothamBold
        confirm.TextSize = 24
        confirm.Text = "Are you sure you want to unload?"
        confirm.Parent = screenGui

        local yesBtn = Instance.new("TextButton")
        yesBtn.Size = UDim2.new(0, 110, 0, 40)
        yesBtn.Position = UDim2.new(0.05, 0, 0.6, 0)
        yesBtn.BackgroundColor3 = Color3.fromRGB(100, 200, 100)
        yesBtn.TextColor3 = Color3.new(1, 1, 1)
        yesBtn.Font = Enum.Font.GothamBold
        yesBtn.TextSize = 20
        yesBtn.Text = "Yes"
        yesBtn.Parent = confirm

        local noBtn = Instance.new("TextButton")
        noBtn.Size = UDim2.new(0, 110, 0, 40)
        noBtn.Position = UDim2.new(0.55, 0, 0.6, 0)
        noBtn.BackgroundColor3 = Color3.fromRGB(200, 100, 100)
        noBtn.TextColor3 = Color3.new(1, 1, 1)
        noBtn.Font = Enum.Font.GothamBold
        noBtn.TextSize = 20
        noBtn.Text = "No"
        noBtn.Parent = confirm

        yesBtn.MouseButton1Click:Connect(function()
            screenGui:Destroy()
        end)
        noBtn.MouseButton1Click:Connect(function()
            confirm:Destroy()
        end)
    end

    local creditsTabBtn = Instance.new("TextLabel")
    creditsTabBtn.Size = UDim2.new(1, 0, 0, 40)
    creditsTabBtn.BackgroundColor3 = Color3.fromRGB(35, 35, 50)
    creditsTabBtn.TextColor3 = Color3.fromRGB(150, 150, 200)
    creditsTabBtn.Font = Enum.Font.GothamItalic
    creditsTabBtn.TextSize = 18
    creditsTabBtn.Text = "Made by iqsm."
    creditsTabBtn.Parent = tabsFrame

    return library
end

return library
